def enviar_formulario(event): event.preventDefault() nome = document.getElementById('nome').value email = document.getElementById('email').value mensagem = document.getElementById('mensagem').value opcao = document.querySelector('input[name="opcao"]:checked').value selecao = document.getElementById('selecao').value interesses = document.getElementById('interesses').selectedOptions console.log("Nome:", nome) console.log("Email:", email) console.log("Mensagem:", mensagem) console.log("Opção:", opcao) console.log("Seleção:", selecao) console.log("Interesses:", [...interesses].map(option => option.value).join(', ')) formulario = document.getElementById('meu-formulario') formulario.addEventListener('submit', enviar_formulario)
